        // Enroll in class
        function enrollClass(classId) {
            if (!currentUser) {
                showNotification('Please login to enroll in classes!', true);
                document.querySelector('[data-section="login"]').click();
                return;
            }
            
            const cls = classes.find(c => c.id === classId);
            if (cls && cls.enrolled < cls.capacity) {
                cls.enrolled++;
                const enrollments = JSON.parse(localStorage.getItem('enrollments') || '[]');
                enrollments.push({
                    userId: currentUser.id,
                    classId,
                    enrolledAt: new Date().toISOString()
                });
                localStorage.setItem('enrollments', JSON.stringify(enrollments));
                showNotification(`Successfully enrolled in ${cls.name}!`);
                loadClasses();
            }
        }
